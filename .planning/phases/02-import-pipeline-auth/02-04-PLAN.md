```typescript
import type { ParticipantRow } from './types'

export interface StoredPreview {
  rows: ParticipantRow[]
  filename: string
  expiresAt: number
}

declare global {
  // eslint-disable-next-line no-var
  var __importPreviewStore: Map<string, StoredPreview> | undefined
}

globalThis.__importPreviewStore ??= new Map<string, StoredPreview>()
const STORE = globalThis.__importPreviewStore

const TTL_MS = 15 * 60 * 1000

function cleanupExpired(): void {
  const now = Date.now()
  for (const [k, v] of STORE.entries()) {
    if (v.expiresAt < now) STORE.delete(k)
  }
}

export function storePreview(token: string, payload: Omit<StoredPreview, 'expiresAt'>): void {
  cleanupExpired()
  STORE.set(token, { ...payload, expiresAt: Date.now() + TTL_MS })
}

export function consumePreview(token: string): StoredPreview | null {
  cleanupExpired()
  const v = STORE.get(token)
  if (!v) return null
  STORE.delete(token)
  return v
}

export function peekPreview(token: string): StoredPreview | null {
  cleanupExpired()
  return STORE.get(token) ?? null
}
```

```typescript
import { NextResponse } from 'next/server'
import { z } from 'zod'
import { requireAdmin } from '@/lib/auth'
import { getSupabase } from '@/lib/supabase'
import { consumePreview } from '@/lib/import/preview-store'

export const runtime = 'nodejs'
export const dynamic = 'force-dynamic'

const Body = z.object({
  serverToken: z.string().min(32).max(64),
  editionYear: z.number().int().optional(),
})

const CHUNK_SIZE = 500

interface RpcResult {
  inserted: number
  updated: number
  errors: number
  error_log: Array<{ excel_row?: number; email?: string; error: string }>
}

export async function POST(req: Request) {
  let admin
  try {
    admin = await requireAdmin()
  } catch {
    return NextResponse.json({ error: 'Forbidden' }, { status: 403 })
  }

  const json = await req.json().catch(() => null)
  const parsed = Body.safeParse(json)
  if (!parsed.success) {
    return NextResponse.json(
      { error: parsed.error.issues[0]?.message ?? 'Body inválido' },
      { status: 400 }
    )
  }
  const { serverToken, editionYear = 2025 } = parsed.data

  const stored = consumePreview(serverToken)
  if (!stored) {
    return NextResponse.json(
      { error: 'Preview expirou ou não encontrado. Faça upload novamente.' },
      { status: 404 }
    )
  }

  const supabase = getSupabase()

  // Resolver edição
  const { data: edition, error: edErr } = await supabase
    .from('editions')
    .select('id')
    .eq('year', editionYear)
    .single()
  if (edErr || !edition) {
    return NextResponse.json(
      { error: `Edição ${editionYear} não encontrada.` },
      { status: 400 }
    )
  }

  // Criar import_jobs PENDING (audit IMPORT-10)
  const { data: job, error: jobErr } = await supabase
    .from('import_jobs')
    .insert({
      edition_id: edition.id,
      filename: stored.filename,
      status: 'PROCESSING',
      total_rows: stored.rows.length,
      imported_by: admin.email,
      inserted_rows: 0,
      updated_rows: 0,
      error_rows: 0,
      error_log: [],
    })
    .select('id')
    .single()

  if (jobErr || !job) {
    return NextResponse.json(
      { error: `Falha ao criar import_job: ${jobErr?.message ?? 'desconhecido'}` },
      { status: 500 }
    )
  }

  // Totais acumulados de participants (RPC primária — usado para inserted_rows/updated_rows/error_rows no import_jobs)
  let totalInserted = 0
  let totalUpdated = 0
  let totalErrors = 0
  // Totais acumulados de form_responses (RPC secundária — somados em error_log e devolvidos no response, mas NÃO contados nas métricas primárias do import_jobs)
  let totalFormInserted = 0
  let totalFormUpdated = 0
  let totalFormErrors = 0
  const allErrorLog: Array<RpcResult['error_log'][0] & { source?: string }> = []

  try {
    // Chunking de 500 linhas (IMPORT-07)
    for (let i = 0; i < stored.rows.length; i += CHUNK_SIZE) {
      const chunk = stored.rows.slice(i, i + CHUNK_SIZE)

      // ─── Payload PARTICIPANTS ───
      // Serializar: a RPC espera p_rows como jsonb. Stringify campos numéricos e booleanos.
      const participantsPayload = chunk.map((r) => ({
        excel_row: r.excel_row,
        email: r.email,
        full_name: r.full_name,
        company: r.company ?? '',
        job_title: r.job_title ?? '',
        cpf: r.cpf ?? '',
        phone: r.phone ?? '',
        payment_status: r.payment_status ?? '',
        is_company_member:
          r.is_company_member === true ? 'true'
          : r.is_company_member === false ? 'false'
          : null,
        company_segment_raw: r.company_segment_raw ?? '',
        company_segment_normalized: r.company_segment_normalized ?? null,
        ticket_membership: r.ticket_membership,
        ticket_value: r.ticket_value !== null ? String(r.ticket_value) : '',
      }))

      const { data: rpcData, error: rpcErr } = await supabase.rpc(
        // @ts-expect-error — função adicionada em Plan 01 Task 3; types serão regenerados.
        'upsert_participants_batch',
        {
          p_rows: participantsPayload,
          p_edition_id: edition.id,
          p_import_job_id: job.id,
        }
      )

      if (rpcErr) {
        throw new Error(`RPC participants erro no chunk ${i}-${i + chunk.length}: ${rpcErr.message}`)
      }
      const result = rpcData as unknown as RpcResult
      totalInserted += result.inserted
      totalUpdated += result.updated
      totalErrors += result.errors
      if (Array.isArray(result.error_log) && result.error_log.length > 0) {
        for (const e of result.error_log) {
          allErrorLog.push({ ...e, source: 'participants' })
        }
      }

      // ─── Payload FORM_RESPONSES ───
      // IMPORTANTE: chamada DEPOIS de participants — RPC resolve participant_id via (email, edition_id)
      // Só envia linhas onde participants foi bem-sucedido OU já existia (não filtra por result.errors
      // porque a RPC de form_responses já tem fallback "Participant não encontrado" para emails órfãos,
      // e queremos preservar essa contagem no error_log).
      const formPayload = chunk.map((r) => ({
        participant_email: r.email,
        excel_row: r.excel_row,
        topics_of_interest: r.topics_of_interest ?? [],
        interested_in_events: r.interested_in_events ?? [],
        preferred_channels: r.preferred_channels ?? [],
        content_interests: r.content_interests ?? [],
        dietary_restrictions: r.dietary_restrictions ?? '',
        dietary_details: r.dietary_details ?? '',
      }))

      const { data: formRpcData, error: formRpcErr } = await supabase.rpc(
        // @ts-expect-error — função adicionada em Plan 01 Task 4; types serão regenerados.
        'upsert_form_responses_batch',
        {
          p_rows: formPayload,
          p_edition_id: edition.id,
        }
      )

      if (formRpcErr) {
        // Não abortar o commit inteiro por falha de form_responses — registrar e seguir.
        // (Participants já foi gravado neste chunk; reverter não é viável sem transação distribuída.)
        allErrorLog.push({
          excel_row: chunk[0]?.excel_row,
          error: `RPC form_responses falhou no chunk ${i}-${i + chunk.length}: ${formRpcErr.message}`,
          source: 'form_responses',
        })
        totalFormErrors += chunk.length
      } else {
        const formResult = formRpcData as unknown as RpcResult
        totalFormInserted += formResult.inserted
        totalFormUpdated += formResult.updated
        totalFormErrors += formResult.errors
        if (Array.isArray(formResult.error_log) && formResult.error_log.length > 0) {
          for (const e of formResult.error_log) {
            allErrorLog.push({ ...e, source: 'form_responses' })
          }
        }
      }
    }

    // Update import_jobs COMPLETED
    // Decisão de modelagem: inserted_rows / updated_rows / error_rows refletem APENAS participants
    // (a entidade primária do import). Contagens de form_responses ficam no error_log + response JSON.
    // Phase 4 (audit log UI) pode renderizar ambas separadamente lendo error_log.
    await supabase
      .from('import_jobs')
      .update({
        status: 'COMPLETED',
        inserted_rows: totalInserted,
        updated_rows: totalUpdated,
        error_rows: totalErrors,
        // @ts-expect-error — error_log é jsonb
        error_log: allErrorLog,
      })
      .eq('id', job.id)

    return NextResponse.json({
      jobId: job.id,
      inserted: totalInserted,
      updated: totalUpdated,
      errors: totalErrors,
      totalRows: stored.rows.length,
      formInserted: totalFormInserted,
      formUpdated: totalFormUpdated,
      formErrors: totalFormErrors,
    })
  } catch (e) {
    // Update import_jobs FAILED
    await supabase
      .from('import_jobs')
      .update({
        status: 'FAILED',
        inserted_rows: totalInserted,
        updated_rows: totalUpdated,
        error_rows: totalErrors,
        // @ts-expect-error — error_log é jsonb
        error_log: [
          ...allErrorLog,
          { error: e instanceof Error ? e.message : 'Erro desconhecido' },
        ],
      })
      .eq('id', job.id)

    return NextResponse.json(
      {
        jobId: job.id,
        error: e instanceof Error ? e.message : 'Erro desconhecido durante commit.',
      },
      { status: 500 }
    )
  }
}
```
